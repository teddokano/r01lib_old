3// - Provides hard fault handler to allow semihosting code not
4// to hang application when debugger not connected.

13// ===== DESCRIPTION =====
14//
15// One of the issues with applications that make use of semihosting operations
16// (such as printf calls) is that the code will not execute correctly when the
17// debugger is not connected. Generally this will show up with the application
18// appearing to just hang. This may include the application running from reset
19// or powering up the board (with the application already in FLASH), and also
20// as the application failing to continue to execute after a debug session is
21// terminated.
22//
23// The problem here is that the "bottom layer" of the semihosted variants of
24// the C library, semihosting is implemented by a "BKPT 0xAB" instruction.
25// When the debug tools are not connected, this instruction triggers a hard
26// fault - and the default hard fault handler within an application will
27// typically just contains an infinite loop - causing the application to
28// appear to have hang when no debugger is connected.
29//
30// The below code provides an example hard fault handler which instead looks
31// to see what the instruction that caused the hard fault was - and if it
32// was a "BKPT 0xAB", then it instead returns back to the user application.
33//
34// In most cases this will allow applications containing semihosting
35// operations to execute (to some degree) when the debugger is not connected.
36//
37// == NOTE ==
38//
39// Correct execution of the application containing semihosted operations
40// which are vectored onto this hard fault handler cannot be guaranteed. This
41// is because the handler may not return data or return codes that the higher
42// level C library code or application code expects. This hard fault handler
43// is meant as a development aid, and it is not recommended to leave
44// semihosted code in a production build of your application!
45//
46// ****************************************************************************
47
48// Allow handler to be removed by setting a define (via command line)
49#if !defined (__SEMIHOST_HARDFAULT_DISABLE)
50
51__attribute__((naked))
52void HardFault_Handler(void){
53 __asm( ".syntax unified\n"
54 // Check which stack is in use
55 "MOVS R0, #4 \n"
56 "MOV R1, LR \n"
57 "TST R0, R1 \n"
58 "BEQ _MSP \n"
59 "MRS R0, PSP \n"
60 "B _process \n"
61 "_MSP: \n"
62 "MRS R0, MSP \n"
63 // Load the instruction that triggered hard fault
64 "_process: \n"
65 "LDR R1,[R0,#24] \n"
66 "LDRH R2,[r1] \n"
67 // Semihosting instruction is "BKPT 0xAB" (0xBEAB)
68 "LDR R3,=0xBEAB \n"
69 "CMP R2,R3 \n"
70 "BEQ _semihost_return \n"
71 // Wasn't semihosting instruction so enter infinite loop
72 "B . \n"
73 // Was semihosting instruction, so adjust location to
74 // return to by 1 instruction (2 bytes), then exit function
75 "_semihost_return: \n"
76 "ADDS R1,#2 \n"
77 "STR R1,[R0,#24] \n"
78 // Set a return value from semihosting operation.
79 // 0 is slightly arbitrary, but appears to allow most
80 // C Library IO functions sitting on top of semihosting to
81 // continue to operate to some degree
82 // Return a positive value (32) for SYS_OPEN only
83 "LDR R1,[ R0,#0 ] \n" // R0 is at location 0 on stack
84 "CMP R1, #1 \n" // 0x01=SYS_OPEN
85 "BEQ _non_zero_ret \n"
86 "MOVS R1,#0 \n"
87 "B _sys_ret \n"
88 "_non_zero_ret: \n"
89 "MOVS R1,#32 \n"
90 "_sys_ret: \n"
91 "STR R1,[ R0,#0 ] \n" // R0 is at location 0 on stack
92 // Return from hard fault handler to application
93 "BX LR \n"
94 ".syntax divided\n") ;
95}

97#endif
