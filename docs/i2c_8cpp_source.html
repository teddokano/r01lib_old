85status_t I2C::write( uint8_t address, const uint8_t *dp, int length, bool stop )
86{
87 status_t r = kStatus_Fail;
88 size_t txCount = 0xFFU;
89
90 if ( (r = LPI2C_MasterStart( EXAMPLE_I2C_MASTER, address, kLPI2C_Write )) )
91 return r;
92
93 LPI2C_MasterGetFifoCounts(EXAMPLE_I2C_MASTER, NULL, &txCount);
94 while (txCount)
95 {
96 LPI2C_MasterGetFifoCounts(EXAMPLE_I2C_MASTER, NULL, &txCount);
97 }
98
99 if ( LPI2C_MasterGetStatusFlags( EXAMPLE_I2C_MASTER ) & kLPI2C_MasterNackDetectFlag )
100 return kStatus_LPI2C_Nak;
101
102 if ( (r = LPI2C_MasterSend( EXAMPLE_I2C_MASTER, (uint8_t *)dp, length )) )
103 {
104 if ( r == kStatus_LPI2C_Nak )
105 LPI2C_MasterStop( EXAMPLE_I2C_MASTER );
106
107 return r;
108 }
109
110 if ( stop )
111 return LPI2C_MasterStop( EXAMPLE_I2C_MASTER );
112
113 return kStatus_Success;
114}

116status_t I2C::read( uint8_t address, uint8_t *dp, int length, bool stop )
117{
118 status_t r = kStatus_Fail;
119 size_t txCount = 0xFFU;
120
121 if ( (r = LPI2C_MasterStart( EXAMPLE_I2C_MASTER, address, kLPI2C_Read )) )
122 return r;
123
124 do {
125 LPI2C_MasterGetFifoCounts( EXAMPLE_I2C_MASTER, NULL, &txCount );
126 } while ( txCount );
127
128 if ( LPI2C_MasterGetStatusFlags( EXAMPLE_I2C_MASTER ) & kLPI2C_MasterNackDetectFlag )
129 return kStatus_LPI2C_Nak;
130
131 if ( (r = LPI2C_MasterReceive( EXAMPLE_I2C_MASTER, dp, length )) )
132 return r;
133
134 if ( stop )
135 return LPI2C_MasterStop( EXAMPLE_I2C_MASTER );
136
137 return kStatus_Success;
138}
