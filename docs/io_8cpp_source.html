69DigitalInOut::DigitalInOut( uint8_t pin_num, bool direction, bool v )
70 : Obj( true ), _pn( pin_num ), _dir( direction ), _value( v )
71{
72 if ( -1 == pins[ _pn ].base )
73 return;
74
75 gpio_pin_config_t led_config = { (gpio_pin_direction_t)direction, _value };
76
77 gpio_n = gpio_type[ pins[ _pn ].base ];
78 port_n = port_type[ pins[ _pn ].base ];
79 gpio_pin = pins[ _pn ].pin;
80
81 GPIO_PinInit( gpio_n, gpio_pin, &led_config );
82 value( (bool)_value );
83}

87void DigitalInOut::value( bool value )
88{
89 if ( -1 == pins[ _pn ].base )
90 return;
91
92 if ( kGPIO_DigitalOutput == _dir )
93 GPIO_PinWrite( gpio_n, gpio_pin, value );
94
95 _value = value;
96}

98bool DigitalInOut::value( void )
99{
100 if ( -1 == pins[ _pn ].base )
101 return 0;
102
103 if ( kGPIO_DigitalInput == _dir )
104 return GPIO_PinRead( gpio_n, gpio_pin );
105 else
106 return _value;
107}
