45 : I2C( sda, scl, true )
46{
47#ifdef CPU_MCXN947VDF
48 if ( (sda == I3C_SDA) && (scl == I3C_SCL) )
49 ;
50 else
51 panic( "FRDM-MCXN947 only support I3C_SDA/I3C_SCL pins for I3C" );
52#else // CPU_MCXN947VDF
53 if ( (sda == I3C_SDA) && (scl == I3C_SCL) )
54 ;
55 else if ( (sda == I2C_SDA) && (scl == I2C_SCL) )
56 ;
57 else
58 panic( "FRDM-MCXA153 supports I3C_SDA/I3C_SCL or I2C_SDA(D18)/I2C_SCL(D19) pins for I3C" );
59#endif // CPU_MCXN947VDF
60
61 I3C_MasterGetDefaultConfig( &masterConfig );
62
63 masterConfig.baudRate_Hz.i2cBaud = i2c_freq ? i2c_freq : I2C::FREQ;
64 masterConfig.baudRate_Hz.i3cOpenDrainBaud = i3c_od_freq ? i3c_od_freq : OD_FREQ;
65 masterConfig.baudRate_Hz.i3cPushPullBaud = i3c_pp_freq ? i3c_pp_freq : PP_FREQ;
66 masterConfig.enableOpenDrainStop = false;
67 masterConfig.disableTimeout = true;
68
69 bus_type = kI3C_TypeI3CSdr;
70
71 I3C_MasterInit( EXAMPLE_MASTER, &masterConfig, I3C_MASTER_CLOCK_FREQUENCY );
72
73 /* Create I3C handle. */
74 I3C_MasterTransferCreateHandle( EXAMPLE_MASTER, &g_i3c_m_handle, &masterCallback, NULL );
75
76 first_broadcast = true;
77

79 DigitalInOut _sda( scl );

81 _scl.pin_mux( kPORT_MuxAlt10 );
82 _sda.pin_mux( kPORT_MuxAlt10 );

173status_t I3C::ccc_broadcast( uint8_t ccc, const uint8_t *dp, uint8_t length, bool first_time )
174{
175 uint8_t bp[ REG_RW_BUFFER_SIZE ];
176 status_t r_code;
177
178 bp[ 0 ] = ccc;
179 memcpy( (uint8_t *)bp + 1, (uint8_t *)dp, length );
180
181 if ( first_time || first_broadcast )
182 {
183 first_broadcast = false;
184
185 frequency( 0, 2000000, 2000000 ); // I2C_freq = default, I3C_OD_freq = 2MHz, I3C_PP_freq = 2MHz
186 r_code = write( BROADCAST_ADDR, bp, length + 1 );
187 frequency(); // revert to default frequency
188 }
189
190 r_code = write( BROADCAST_ADDR, bp, length + 1 );
191
192 return r_code;
193}

225void I3C::master_ibi_callback( I3C_Type *base, i3c_master_handle_t *handle, i3c_ibi_type_t ibiType, i3c_ibi_state_t ibiState )
226{
227 g_ibiWonFlag = true;
228 g_ibiAddress = handle->ibiAddress;
229
230 switch ( ibiType )
231 {
232 case kI3C_IbiNormal:
233 if ( ibiState == kI3C_IbiDataBuffNeed )
234 {
235 handle->ibiBuff = g_ibiBuff;
236 }
237 else
238 {
239 memcpy( g_ibiUserBuff, (void *)handle->ibiBuff, handle->ibiPayloadSize );
240 g_ibiUserBuffUsed = handle->ibiPayloadSize;
241 }
242 break;
243
244 default:
245 assert(false);
246 break;
247 }
248
249 if ( g_ibi_callback )
250 g_ibi_callback();
251}
