56 /* SPI */

59
60 SYSCON->CLOCK_CTRL |= SYSCON_CLOCK_CTRL_FRO1MHZ_ENA_MASK; // UTICK
61
62 CLOCK_EnableClock( kCLOCK_Gpio0 );
63 CLOCK_EnableClock( kCLOCK_Gpio1 );
64 CLOCK_EnableClock( kCLOCK_Gpio3 );
65 CLOCK_EnableClock( kCLOCK_Gpio4 );
66
67 /* Init board hardware. */
68 BOARD_InitBootPins();
69 BOARD_InitBootClocks();
70 BOARD_InitBootPeripherals();
71 #ifndef BOARD_INIT_DEBUG_CONSOLE_PERIPHERAL
72 /* Init FSL debug console. */

77 /* Attach clock to I3C 24MHZ */
