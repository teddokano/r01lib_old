38void irq_handler( int num )
39{
40 uint32_t flags;
41 flags = GPIO_GpioGetInterruptFlags( gpio_ptr[ num ] );
42 GPIO_GpioClearInterruptFlags( gpio_ptr[ num ], flags );
43
44 for ( int i = 0; i < GPIO_BITS; i++ )
45 if ( cb_table[ num ][ i ] && (flags & (1 << i)) )
46 (cb_table[ num ][ i ])();
47}
